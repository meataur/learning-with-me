```mermaid
mindmap
  root((Determiners))
    (Pre-determiners)
      "Rule: Appear *before* central determiners"
      (Multipliers)
        "double, twice, three times"
        "Ex: `twice the effort`, `double the pay`"
      (Fractions)
        "half, one-third, a quarter"
        "Ex: `half a cup`, `a third of the pie`"
      (Intensifiers)
        "what, such, rather, quite"
        "Ex: `what a surprise!`, `such a mess`"
      (All / Both)
        "all, both"
        "Ex: `all the people`, `both my hands`"
    (Central Determiners)
      "Rule: The most common type; only one per noun phrase"
      (Articles)
        (Definite: the)
          "Usage: Specific, unique nouns"
          "Nouns: Singular, Plural, Uncountable"
          "Ex: `the moon`, `the books`, `the water`"
        (Indefinite: a/an)
          "Usage: General, non-specific nouns"
          "Nouns: Singular Countable only"
          "Ex: `a dog`, `an hour`, `a university`"
      (Possessives)
        "my, your, his, her, its, our, their"
        "Usage: Shows ownership"
        "Ex: `my car`, `her idea`, `John's house`"
      (Demonstratives)
        "this, that (Singular/Uncountable)"
        "these, those (Plural)"
        "Usage: Points to nouns by proximity"
        "Ex: `this chair`, `those stars`"
      (Interrogatives)
        "which, what, whose"
        "Usage: Asks questions about nouns"
        "Ex: `Which path?`, `What time?`"
    (Post-determiners)
      "Rule: Appear *after* central determiners"
      (Cardinal Numbers)
        "one, two, three..."
        "Ex: `the two cats`, `my one regret`"
      (Ordinal Numbers)
        "first, second, third..."
        "Ex: `my first attempt`, `the second chapter`"
      (General Ordinals)
        "next, last, previous, subsequent"
        "Ex: `the next station`, `the last person`"
      (Quantifiers)
        "many, few, several, little, much"
        "Ex: `the many reasons`, `a little patience`"
```
